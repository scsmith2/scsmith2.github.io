Dataset Preparation

The inital dataset will be divided into a training set (75%) and a testing set (25%). The training set is used to build the predictive models and the testing set is used to gauge out-of-sample error and thereby, goodness of the model. Root-Mean Squared Error (RMSE) will be used as a metric to gauge goodness of the model.

set.seed(3)
inTrain <- createDataPartition(data$mpg, p = 3/4, list = FALSE)
train <- data [inTrain,]; test <- data[-inTrain,]

Simple Linear Model [mpg = f(displacement)]

Fitting a Linear Model for fuel efficiency as and calculating the RMSE on predictions for the testing set.

lmFit <- lm(mpg ~ disp, data = train)

plot(data$disp, data$mpg, type = "p",
     xlab = "Displacement [in^3]",
     ylab = "Fuel Efficiency [mpg]")
abline(coefficients(lmFit)); grid(col = "gray50")

lmPred <- predict(lmFit, test)
plot(test$mpg, lmPred, xlab = "Tested mpg", ylab = "Predicted mpg", xlim = c(10,35), ylim = c(10,35),
     main = "Simple Linear Model Predictions"); abline(0,1); grid(col = "gray50")

lmRMSE <- (1/length(test$mpg)*sum((lmPred - test$mpg)^2))^.5

Generalized Linear Model

Fitting a linear model combining all features and returning RMSE for test set predictions, as well as the individual coefficients.

glmFit <- train(mpg ~ ., data = train, method = "glm")
glmPred <- predict(glmFit, test)
plot(test$mpg, glmPred, xlab = "Tested mpg", ylab = "Predicted mpg", xlim = c(10,35), ylim = c(10,35),
     main = "Generalized Linear Model Predictions"); abline(0,1); grid(col = "gray50")

glmRMSE <- (1/length(test$mpg)*sum((glmPred - test$mpg)^2))^.5
print(coef(glmFit$finalModel))
